def on_const(tok, data)
  case @lex_state
  when EXPR_FNAME
    @lex_state = EXPR_ENDFN
  when EXPR_CLASS, EXPR_CMDARG, EXPR_MID
    @lex_state = EXPR_ARG
  else
    @lex_state = EXPR_CMDARG
  end
  data << Token.new(lineno, column, __method__, tok, @lex_state)
end
